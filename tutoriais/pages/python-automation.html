⚠️ Erro Comum #1: Caminhos de arquivo incorretos

❌ Erro: Usar caminhos fixos com barras 
# Código quebradiço - só funciona no Windows
with open('C:\Users\joao\Documents\arquivo.txt', 'r') as f:
    conteudo = f.read()

# Problema: \n, \t são interpretados como caracteres especiais
# O código quebra em Linux/Mac e com nomes de usuário diferentes

Problema: Código não é portável e usa caracteres de escape incorretamente.

✅ Correção: Usar pathlib ou os.path 
from pathlib import Path
import os

# Método 1: pathlib (recomendado)
caminho = Path.home() / 'Documents' / 'arquivo.txt'

# Método 2: os.path
caminho = os.path.join(os.path.expanduser('~'), 'Documents', 'arquivo.txt')

with open(caminho, 'r') as f:
    conteudo = f.read()

# pathlib também oferece métodos úteis:
if caminho.exists():
    print(f"Arquivo existe: {caminho.stat().st_size} bytes")

⚠️ Erro Comum #2: Não tratar erros em operações de arquivo

❌ Erro: Assumir que o arquivo sempre existe 
import shutil

# Se o arquivo não existir, o programa quebra
shutil.move('origem.txt', 'destino/')  # FileNotFoundError

Problema: O script falha silenciosamente ou interrompe a execução.

✅ Correção: Verificar existência e tratar exceções 
import shutil
from pathlib import Path

def mover_arquivo_seguro(origem, destino):
    """
    Move arquivo de forma segura, criando diretórios se necessário
    """
    origem_path = Path(origem)
    destino_path = Path(destino)
    
    if not origem_path.exists():
        print(f"Erro: Arquivo {origem} não encontrado")
        return False
    
    try:
        # Criar diretório de destino se não existir
        destino_path.parent.mkdir(parents=True, exist_ok=True)
        
        # Mover arquivo
        shutil.move(str(origem_path), str(destino_path))
        print(f"Arquivo movido com sucesso para {destino_path}")
        return True
    except Exception as e:
        print(f"Erro ao mover arquivo: {e}")
        return False

# Uso seguro
mover_arquivo_seguro('origem.txt', 'destino/backup/origem.txt')

⚠️ Erro Comum #3: Web scraping sem respeitar o robots.txt

❌ Erro: Fazer scraping agressivo sem limites 
import requests
from bs4 import BeautifulSoup

# Scraping sem controle - pode sobrecarregar o servidor
for i in range(1000):
    response = requests.get(f'https://site.com/page/{i}')
    # Processar dados...
    # Sem delay, sem verificar robots.txt

Problema: Pode causar sobrecarga no servidor, ser bloqueado ou até mesmo ser ilegal.

✅ Correção: Implementar boas práticas de scraping 
import requests
import time
from bs4 import BeautifulSoup
from urllib.robotparser import RobotFileParser

class ScraperRespeitoso:
    def __init__(self, base_url, delay=1):
        self.base_url = base_url
        self.delay = delay
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'MeuBot-Aprendizado/1.0'
        })
        self.verificar_robots()
    
    def verificar_robots(self):
        """Verifica se o scraping é permitido"""
        rp = RobotFileParser()
        rp.set_url(f"{self.base_url}/robots.txt")
        rp.read()
        self.robot_parser = rp
    
    def pode_scraping(self, url):
        """Verifica se a URL pode ser acessada"""
        return self.robot_parser.can_fetch('*', url)
    
    def get_page(self, url):
        """Busca página com delay respeitoso"""
        if not self.pode_scraping(url):
            print(f"Scraping não permitido para: {url}")
            return None
        
        time.sleep(self.delay)  # Delay respeitoso
        try:
            response = self.session.get(url)
            response.raise_for_status()
            return response.text
        except requests.RequestException as e:
            print(f"Erro ao acessar {url}: {e}")
            return None

# Uso
scraper = ScraperRespeitoso('https://exemplo.com', delay=2)
html = scraper.get_page('https://exemplo.com/pagina')

import os
import shutil
from pathlib import Path

def organizar_downloads():
    """Organiza arquivos da pasta Downloads por tipo"""
    pasta_downloads = Path.home() / 'Downloads'
    
    # Mapeamento de extensões para pastas
    categorias = {
        'imagens': ['.jpg', '.jpeg', '.png', '.gif', '.bmp'],
        'documentos': ['.pdf', '.docx', '.txt', '.xlsx', '.pptx'],
        'programas': ['.exe', '.msi', '.dmg', '.deb'],
        'compactados': ['.zip', '.rar', '.7z', '.tar.gz'],
        'codigo': ['.py', '.js', '.html', '.css', '.json']
    }
    
    for arquivo in pasta_downloads.iterdir():
        if arquivo.is_file():
            extensao = arquivo.suffix.lower()
            
            for categoria, extensoes in categorias.items():
                if extensao in extensoes:
                    pasta_destino = pasta_downloads / categoria
                    pasta_destino.mkdir(exist_ok=True)
                    
                    destino = pasta_destino / arquivo.name
                    shutil.move(str(arquivo), str(destino))
                    print(f"Movido: {arquivo.name} -> {categoria}/")
                    break

if __name__ == "__main__":
    organizar_downloads()

2. Envio automático de relatórios (Intermediário)

import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email import encoders
import pandas as pd
from datetime import datetime

def enviar_relatorio_email(destinatario, assunto, corpo, anexo=None):
    """
    Envia email com relatório usando SMTP (Gmail)
    """
    # Configurações (use variáveis de ambiente para senhas!)
    remetente = "seuemail@gmail.com"
    senha = "sua_senha_app"  # Use senha de app do Gmail
    
    # Criar mensagem
    msg = MIMEMultipart()
    msg['From'] = remetente
    msg['To'] = destinatario
    msg['Subject'] = assunto
    
    # Corpo do email
    msg.attach(MIMEText(corpo, 'html'))
    
    # Anexo
    if anexo:
        with open(anexo, 'rb') as f:
            part = MIMEBase('application', 'octet-stream')
            part.set_payload(f.read())
            encoders.encode_base64(part)
            part.add_header(
                'Content-Disposition',
                f'attachment; filename={anexo.split("/")[-1]}'
            )
            msg.attach(part)
    
    # Enviar
    try:
        server = smtplib.SMTP('smtp.gmail.com', 587)
        server.starttls()
        server.login(remetente, senha)
        server.send_message(msg)
        server.quit()
        print("Email enviado com sucesso!")
        return True
    except Exception as e:
        print(f"Erro ao enviar email: {e}")
        return False

def gerar_relatorio_vendas():
    """Gera relatório em HTML a partir de dados"""
    # Dados de exemplo
    dados = {
        'produto': ['Notebook', 'Mouse', 'Teclado'],
        'vendas': [150, 300, 200],
        'receita': [45000, 15000, 10000]
    }
    
    df = pd.DataFrame(dados)
    
    html = f"""
    
    
        
    
    
        
Relatório de Vendas - {datetime.now().strftime('%d/%m/%Y')}

        {df.to_html(index=False)}
        
Total de Vendas: {df['vendas'].sum()}

        
Receita Total: R$ {df['receita'].sum():,.2f}

    
    
    """
    
    with open('relatorio_vendas.html', 'w') as f:
        f.write(html)
    
    return 'relatorio_vendas.html'

# Uso
arquivo_relatorio = gerar_relatorio_vendas()
enviar_relatorio_email(
    'gestor@empresa.com',
    'Relatório Diário de Vendas',
    'Segue em anexo o relatório de vendas do dia.',
    arquivo_relatorio
)

from bs4 import BeautifulSoup
import time
import json
from datetime import datetime

class MonitorPrecos:
    def __init__(self, config_file='config.json'):
        with open(config_file) as f:
            self.config = json.load(f)
        self.historico = {}
    
    def verificar_preco(self, url, seletor):
        """Verifica preço de um produto"""
        try:
            headers = {
                'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
            }
            response = requests.get(url, headers=headers, timeout=10)
            soup = BeautifulSoup(response.text, 'html.parser')
            
            elemento = soup.select_one(seletor)
            if elemento:
                preco_texto = elemento.text.strip()
                # Limpar texto para extrair número
                import re
                numeros = re.findall(r'[\d.,]+', preco_texto)
                if numeros:
                    preco = float(numeros[0].replace('.', '').replace(',', '.'))
                    return preco
        except Exception as e:
            print(f"Erro ao verificar {url}: {e}")
        return None
    
    def salvar_historico(self, produto, preco):

💡 Dicas Avançadas

  • Agendamento: Use schedule para scripts Python ou cron (Linux/Mac) / Task Scheduler (Windows)

⚠️ Atenção Legal:
