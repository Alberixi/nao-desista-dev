📌 Classe e Objeto

class Usuario {
    private $nome;
    private $email;
    
    public function __construct($nome, $email) {
        $this->nome = $nome;
        $this->email = $email;
    }
    
    public function getNome() {
        return $this->nome;
    }
}

$usuario = new Usuario("João", "joao@email.com");
echo $usuario->getNome();
