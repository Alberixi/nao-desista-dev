📌 Decorators

def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"Tempo: {time.time()-start:.2f}s")
        return result
    return wrapper

@timer
def operacao_demorada():
    time.sleep(2)
